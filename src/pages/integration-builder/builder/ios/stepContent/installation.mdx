import Tabs from "@theme/Tabs";

Install Web3Auth Swift SDK using either Cocoapods or Swift Package Manager:

<Tabs
  defaultValue="pods"
  values={[
    { label: "Cocoapods", value: "pods" },
    { label: "Swift Package Manager", value: "spm" },
  ]}
>
<TabItem value="pods">

```
  pod 'Web3Auth'
```

</TabItem>

<TabItem value="spm">

**File > Swift Packages > Add Package Dependency...**

In the Choose Package Repository prompt add this url:

```
https://github.com/Web3Auth/web3auth-swift-sdk/releases/tag/6.0.1
```

</TabItem>

</Tabs>

:::warning Update to the Latest-SDK

Please use the **[latest version](https://github.com/Web3Auth/web3auth-swift-sdk/releases)** of Web3Auth's iOS SDK.

:::
