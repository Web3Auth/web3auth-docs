import Tabs from "@theme/Tabs";

Install `@web3auth/web3auth`, `@web3auth/base` & `@web3auth/openlogin-adapter` using either [npm](https://www.npmjs.com/package/@web3auth/web3auth) or
[yarn](https://yarnpkg.com/package/@web3auth/web3auth):

<Tabs
  defaultValue="npm"
  values={[
    { label: "npm", value: "npm" },
    { label: "yarn", value: "yarn" },
  ]}
>
<TabItem value="npm">

```shell
npm install --save @web3auth/web3auth @web3auth/base @web3auth/openlogin-adapter
```

</TabItem>

<TabItem value="yarn">

```shell
yarn add @web3auth/web3auth @web3auth/base @web3auth/openlogin-adapter
```

</TabItem>

</Tabs>
