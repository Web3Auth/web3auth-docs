import Tabs from "@theme/Tabs";

Install CustomAuth Web SDK using either [npm](https://www.npmjs.com/package/@toruslabs/customauth) or CDN:

<Tabs
  defaultValue="npm"
  values={[
    { label: "npm", value: "npm" },
    { label: "jsdelivr", value: "jsdelivr" },
  ]}
>
<TabItem value="npm">

```shell
npm i --save @toruslabs/customauth
```

</TabItem>

<TabItem value="jsdelivr">

```html
<script src="https://cdn.jsdelivr.net/npm/@toruslabs/customauth@6/dist/directWebSdk.umd.min.js"></script>
```

</TabItem>

</Tabs>
