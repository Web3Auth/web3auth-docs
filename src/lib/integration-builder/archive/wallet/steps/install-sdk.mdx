---

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

Install Torus Embed SDK using `npm` or `yarn`:

<Tabs
  defaultValue="npm"
  values={[
    { label: "npm", value: "npm" },
    { label: "Yarn", value: "yarn" },
  ]}
>
<TabItem value="npm">

```shell
npm i --save @toruslabs/torus-embed
```

</TabItem>

<TabItem value="yarn">

```shell
yarn add @toruslabs/torus-embed
```

</TabItem>

</Tabs>
