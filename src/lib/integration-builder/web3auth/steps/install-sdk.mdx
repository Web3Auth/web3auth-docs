Install OpenLogin SDK using either [npm](https://www.npmjs.com/package/@toruslabs/openlogin) or CDN:

<Tabs
  defaultValue="npm"
  values={[
    { label: "npm", value: "npm" },
    { label: "jsdelivr", value: "jsdelivr" },
  ]}
>
<TabItem value="npm">

```shell
npm i --save @toruslabs/openlogin
```

</TabItem>

<TabItem value="jsdelivr">

```html
<script src="https://cdn.jsdelivr.net/npm/@toruslabs/openlogin@1"></script>
```

</TabItem>

</Tabs>
