### Get the User Profile

```tsx
const parseToken = (token: any) => {
  try {
    const base64Url = token.split(".")[1];
    const base64 = base64Url.replace("-", "+").replace("_", "/");
    return JSON.parse(window.atob(base64 || ""));
  } catch (err) {
    console.error(err);
    return null;
  }
};

const userProfile = parseToken(idToken);
```
