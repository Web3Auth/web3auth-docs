```ts
interface UIConfig {
  /**
   * Logo for your app.
   */
  appLogo?: string;
  /**
   * theme for the modal
   *
   * @defaultValue `light`
   */
  theme?: "light" | "dark";
  /**
   * order of how login methods are shown
   *
   * @defaultValue `["google", "facebook", "twitter", "reddit", "discord", "twitch", "apple", "line", "github", "kakao", "linkedin", "weibo", "wechat", "email_passwordless"]`
   */
  loginMethodsOrder?: string[];
}
```
